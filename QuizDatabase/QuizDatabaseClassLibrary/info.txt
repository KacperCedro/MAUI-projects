Table answers {
  Id integer [primary key]
  answerContent varchar
  isCorrect bool
  questionId integer
}

Table users {
  Id integer [primary key]
  username varchar
  password varchar
}

Table questions {
  Id integer [primary key]
  content varchar
}

Table answersUsers {
  Id int [primary key]
  answerId int
  userId int
}

Ref: answers.questionId > questions.Id // many-to-one

Ref: answersUsers.answerId < answers.Id

Ref: answersUsers.userId < users.Id
